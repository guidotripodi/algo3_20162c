A continuaci\'on se detalla el pseudo-c\'odigo de la parte principal del algoritmo:

\begin{algorithm}[H]
\caption{Mochilas}
\begin{algorithmic}[1]
\Function{Algoritmo}{\param{in}{Integer}{M}\param{in}{Integer}{N}}{$\ensuremath{\rightarrow}$ \param{out}{S} {Integer} \param{out}{elementos} {List<Integer>}}
\If{M == 1} \hfill //O(1)
\state maximizoSimple(1) \hfill //O(M * N)
\endif
\If{M == 2} \hfill //O(1)
\If{capacidad[M${_1}$] == capacidad[M${_2}$]} \hfill //O(1)
\state maximizoSimple(1) \hfill //O(M * N)
\state maximizoSimple(2) \hfill //O(M * N)
\Else
\state maximizoDoble(1,2) \hfill //O(M * N)
\endif
\endif
\If{M == 3} \hfill //O(1)
\If{capacidad[M${_1}$] == capacidad[M${_2}$] $\wedge$ capacidad[M${_2}$] == capacidad[M${_3}$] } \hfill //O(1)
\state maximizoSimple(1) \hfill //O(M * N)
\state maximizoSimple(2) \hfill //O(M * N)
\state maximizoSimple(3) \hfill //O(M * N)
\Else
\state maximizoTriple(1,2,3) \hfill //O(M * N)
\endif
\endif
\state devolver arrayMaximo.size \hfill //O(N)
\state devolver arrayMaximo \hfill //O(N)
\EndFunction 
\end{algorithmic}
\hrule
\complejidad{$M^3 \ast N$}
\end{algorithm}
