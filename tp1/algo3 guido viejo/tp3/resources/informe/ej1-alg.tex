
\begin{algorithm}[H]
\caption{}
\begin{algorithmic}[1]
\Function{Solve}{\param{in}{portals}{List<List< <Boolean> >}}{$\ensuremath{\rightarrow}$ \param{out}{res}{List<Integer>}}
\state maxPortal $\gets$ int[size(portals)]
\state maxPortal[0] $\gets$ 0
\state fill(maxPortal, MIN-INTEGER)
\state i $\gets$ 1
\state j $\gets$ 0
\while{i < size(maxPortal)}
	\while{j < i}
		\If{portals[i][j]}
			\state maxPortal[i] $\gets$ max(maxPortal[i],maxPortal[j] + 1)
		\endif
		\state j++
	\endwhile
	\state i++
\endwhile
\state res $\gets$ maxPortal[size(maxPortal) - 1]
\EndFunction 
\end{algorithmic}
\hrule
\complejidad{$N^2$}
\end{algorithm}
