 \caption{Algoritmo EJ1}

	entero sumaParcial $\gets$ 0 \\
	arreglo sumasParciales \\
    \While{sumaParcial $\leq$ P}{
      sumasParciales $\cup$ sumaParcial\\
	para sumaParcial asignar sumaParcial + 3$^{i}$\\
     	incrementar i \\
     }
	entero indice $\gets$ size(sumasParciales)-1   \\
	entero equilibrioActual $\gets$ P \\
	entero potenciaActual $\gets$ 0 \\
	    \While{|equilibrioActual| $>$ 0}{
	          \eIf{sumasParciales[indice-1] $<$ |equilibrioActual| $\leq$ sumasParciales[indice]}{
       para potenciaActual asignar sumasParciales[indice] -  sumasParciales[indice-1]\\
       	\eIf{equilibrioActual $<$ 0}{
       	   	para equilibrioActual asignar potenciaActual + equilibrioActual \\
                platoDerecho $\cup$ potenciaActual \\
			}{
			para equilibrioActual asignar equilibrioActual - potenciaActual \\
                platoIzquierdo $\cup$ potenciaActual \\
      }
     }
     indice--	
	}
	devolver size(platoDerecho) \\      
		devolver size(platoIzquierdo) \\      
			devolver invertir(platoDerecho) \\      
				devolver invertir(platoIzquierdo) \\      
     


\end{algorithm}
