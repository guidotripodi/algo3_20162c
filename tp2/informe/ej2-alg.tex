	    aristas.push(a)\\
	    incrementar cantidadAristas en 1\\
	    }
	    nodo derecha asignar mapa[elemento.i][elemento.j+1]\\
	    \If{valor(derecha) $\neq$ invalido}{

	    para a.fin asignar (elemento.i $\ast$ C) + (elemento.j + 2)
	    }
	    para a.costo asignar valor(derecha)\\
	    incrementar cantidadAristas en 1\\
	    }
	    }	   
	    ordenar(aristas)\\
	    \If{find(a.inicio) $\neq$ find(a.fin)}{

\end{algorithm}


\textbf{Aclaraciones de variables y transformaci\'on de entrada}

\end{itemize}
