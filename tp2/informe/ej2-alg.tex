\begin{algorithm}[H] %or another one check
 \Fn{Kruskal}{
 %     \SetAlgoLined
     ordenar aristas \hfill O(4 $\ast$ F $\ast$ C $\ast$ log(4 $\ast$ F $\ast$ C))\\
	representanteFinal $\gets$ -1  \hfill O(1)
	
	\For{Arista ar $\in$ aristas}{ 
					\hfill  ciclo: O(4 $\ast$ F $\ast$ C)\\
		\If{Buscar(ar.inicio) = Buscar(ar.fin)}{
									 \hfill guarda: O(2 $\ast$ log(4 $\ast$ F $\ast$ C))\\
			Unir(ar.inicio, ar.fin, ar.costo) \hfill O(1)
		}

		\If{representanteFinal $\geq$ 0}{
			finalizar ciclo \hfill O(1)
		}
	}

	\If{representanteFinal $\geq$ 0}{
		 devolver costCompLider[representanteFinal] \hfill O(1)
	}\Else{
         devolver sinSolucion \hfill O(1)
    }
  }
 
  \textbf{\hfill total: O(F $\ast$ C $\ast$ log(F $\ast$ C))}\\
\end{algorithm}


\begin{algorithm}[H] %or another one check
 \Fn{Buscar(entero x)}{
 %     \SetAlgoLined
 	\If{padre[x] = x}{devolver x \hfill O(1)}{
 	para entero p $\gets$ Buscar(padre[x]) \hfill guarda: O(log(4 $\ast$ F $\ast$ C))\\
	para padre[x] asignar p \hfill O(1)\\
	devolver p\hfill O(1)\\
 	}
 	}
 	
 	\textbf{\hfill total: O(log(4 $\ast$ F $\ast$ C))}\\
\end{algorithm}


\begin{algorithm}[H] %or another one check
 \Fn{Unir(enteros x, y, costo)}{
 %     \SetAlgoLined
 	para entero x asignar Buscar(X)\hfill O(1)\\
 	para entero y asignar Buscar(Y)\hfill O(1)\\
	\eIf{altura[x] $\leq$ altura[y]} {
	 	para padre[x] asignar y \hfill O(1)\\
 	para cantAristas[y] asignar cantAristas[y] + cantAristas[x] +1\hfill O(1)\\
 	para costCompLider[y] asignar costCompLider[y] + costCompLider[x] + costo\hfill O(1)\\
 	\If{altura[x] = altura[y]}{
 	 	para altura[x] incrementar 1 \hfill O(1)}
 	 	\If {cantAristas[y] = $\#$nodos-1} {
            para representanteFinal asignar y\hfill O(1)\\
        }
	}{ 	 	para padre[y] asignar x\hfill O(1)\\
 	para cantAristas[x] asignar cantAristas[y] + cantAristas[x] +1\hfill O(1)\\
 	para costCompLider[x] asignar costCompLider[x] + costCompLider[y] + costo\hfill O(1)\\
 	 \If{cantAristas[x] = $\#$nodos-1} {
           para representanteFinal asignar x\hfill O(1)\\
        }	}
 }
  	\textbf{\hfill total: O(1)}\\
\end{algorithm}

\textbf{Aclaraciones de variables y transformaci\'on de entrada}

\begin{itemize}

\item {\bf Arreglo cantAristas}: guarda la cantidad de aristas que tiene el representante de una componente conexa.
\item {\bf Arreglo costCompLider}: guarda el costo total de una componente conexa por representante. 
\item {\bf Arreglo padre}: para cada nodo, quien es su representante, es decir, en que componente conexa se encuentra.
\item {\bf Arreglo altura}: indica la altura de cada componente conexa.
\end{itemize}
